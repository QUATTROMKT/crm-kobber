import React, { useState, useEffect } from 'react';
import { 
  Save, 
  User, 
  Phone, 
  MapPin, 
  Wrench, 
  Car, 
  Search, 
  DollarSign, 
  XCircle, 
  CheckCircle, 
  ClipboardList,
  AlertCircle,
  Briefcase,
  Download,
  LogOut,
  Shield
} from 'lucide-react';
import { initializeApp } from 'firebase/app';
import { 
  getFirestore, 
  collection, 
  addDoc, 
  serverTimestamp, 
  query, 
  orderBy, 
  limit, 
  onSnapshot,
  getDocs 
} from 'firebase/firestore';
import { 
  getAuth, 
  signInAnonymously, 
  onAuthStateChanged,
  signInWithCustomToken,
  signOut
} from 'firebase/auth';

// --- CONFIGURAÇÃO FIREBASE ---
const firebaseConfig = JSON.parse(__firebase_config);
const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);
const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';

// --- CONSTANTES ---
const ESTADOS_FOCO = ["RS", "SC", "PR", "SP", "Outro"];

const ORIGENS_CLIENTE = [
  "Redes Sociais (Insta/Face)",
  "Google (Pesquisa)",
  "Google Maps (Meu Negócio)",
  "OLX / Marketplace",
  "Indicação",
  "Já era cliente",
  "Passante (Frente loja)"
];

const MOTIVOS_PERDA = [
  "Falta de Estoque",
  "Preço (Concorrência)",
  "Prazo/Frete",
  "Só pesquisando/Curioso",
  "Cliente vai pensar"
];

// --- COMPONENTES UI AUXILIARES ---

const SectionTitle = ({ icon: Icon, title }) => (
  <div className="flex items-center gap-2 mb-4 border-b pb-2 border-slate-200 mt-6">
    <Icon className="w-5 h-5 text-blue-600" />
    <h3 className="font-bold text-slate-700 uppercase text-sm tracking-wide">{title}</h3>
  </div>
);

const SelectButton = ({ selected, options, onSelect }) => (
  <div className="flex flex-wrap gap-2">
    {options.map((opt) => {
      const isSelected = selected === opt;
      return (
        <button
          key={opt}
          type="button"
          onClick={() => onSelect(opt)}
          className={`px-4 py-2 rounded-lg text-sm font-medium transition-all duration-200 border ${
            isSelected
              ? 'bg-blue-600 text-white border-blue-600 shadow-md transform scale-105'
              : 'bg-white text-slate-600 border-slate-300 hover:bg-slate-50'
          }`}
        >
          {opt}
        </button>
      );
    })}
  </div>
);

// --- TELA DE LOGIN SIMPLES ---
const LoginScreen = ({ onLogin }) => {
  const [email, setEmail] = useState('');
  const [nome, setNome] = useState('');
  const [isAdmin, setIsAdmin] = useState(false);

  const handleLogin = (e) => {
    e.preventDefault();
    if (!nome || !email) return alert("Preencha nome e email");
    
    // Simulação simples de roles baseada no email (para facilitar o deploy inicial)
    // Em produção, usaríamos Firebase Auth com senha.
    const role = (email.includes('admin') || isAdmin) ? 'admin' : 'vendedor';
    onLogin({ name: nome, email: email, role: role });
  };

  return (
    <div className="min-h-screen bg-blue-900 flex items-center justify-center p-4">
      <div className="bg-white rounded-xl shadow-2xl p-8 max-w-md w-full">
        <div className="text-center mb-8">
          <div className="bg-blue-100 w-16 h-16 rounded-full flex items-center justify-center mx-auto mb-4">
            <Car className="w-8 h-8 text-blue-800" />
          </div>
          <h1 className="text-2xl font-bold text-slate-800">Kobber CRM</h1>
          <p className="text-slate-500">Identifique-se para acessar</p>
        </div>

        <form onSubmit={handleLogin} className="space-y-4">
          <div>
            <label className="block text-sm font-medium text-slate-700 mb-1">Seu Nome Completo</label>
            <input 
              type="text" 
              required
              className="w-full p-3 border border-slate-300 rounded-lg focus:ring-2 focus:ring-blue-500 outline-none"
              placeholder="Ex: João Silva"
              value={nome}
              onChange={e => setNome(e.target.value)}
            />
          </div>
          
          <div>
            <label className="block text-sm font-medium text-slate-700 mb-1">Email Corporativo</label>
            <input 
              type="email" 
              required
              className="w-full p-3 border border-slate-300 rounded-lg focus:ring-2 focus:ring-blue-500 outline-none"
              placeholder="seu.nome@kobber.com"
              value={email}
              onChange={e => setEmail(e.target.value)}
            />
          </div>

          <div className="flex items-center gap-2 py-2">
            <input 
              type="checkbox" 
              id="adminCheck"
              className="w-4 h-4 text-blue-600 rounded"
              checked={isAdmin}
              onChange={e => setIsAdmin(e.target.checked)}
            />
            <label htmlFor="adminCheck" className="text-sm text-slate-600">Sou Administrador (Controle)</label>
          </div>

          <button 
            type="submit"
            className="w-full bg-blue-800 hover:bg-blue-900 text-white font-bold py-3 rounded-lg transition-colors"
          >
            ENTRAR NO SISTEMA
          </button>
        </form>
      </div>
    </div>
  );
};

// --- COMPONENTE PRINCIPAL ---

export default function App() {
  const [firebaseUser, setFirebaseUser] = useState(null);
  const [appUser, setAppUser] = useState(null); // Usuário logado na aplicação (Nome/Role)
  const [loading, setLoading] = useState(false);
  const [successMsg, setSuccessMsg] = useState('');
  const [recentOpps, setRecentOpps] = useState([]);
  const [exporting, setExporting] = useState(false);

  // Estado do Formulário
  const initialFormState = {
    clienteNome: '',
    clienteTelefone: '',
    clienteEmail: '',
    clienteCidade: '',
    clienteUF: 'RS',
    tipoCliente: 'Consumidor Final', 
    oficinaNome: '',
    oficinaFoco: '', // Agora é texto livre
    pecaProcurada: '',
    veiculoModelo: '',
    origem: '',
    houveVenda: null,
    valorVenda: '',
    motivoPerda: '',
    pecaFaltante: '',
    observacoes: ''
  };

  const [formData, setFormData] = useState(initialFormState);

  // --- AUTENTICAÇÃO FIREBASE ---
  useEffect(() => {
    const initAuth = async () => {
      if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
        await signInWithCustomToken(auth, __initial_auth_token);
      } else {
        await signInAnonymously(auth);
      }
    };
    initAuth();
    const unsubscribe = onAuthStateChanged(auth, (u) => {
      setFirebaseUser(u);
    });
    return () => unsubscribe();
  }, []);

  // --- CARREGAR DADOS RECENTES ---
  useEffect(() => {
    if (!firebaseUser) return;

    const q = query(
      collection(db, 'artifacts', appId, 'public', 'data', 'opportunities'),
      orderBy('createdAt', 'desc'),
      limit(5)
    );

    const unsubscribe = onSnapshot(q, (snapshot) => {
      const data = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
      setRecentOpps(data);
    }, (error) => console.error("Erro busca:", error));

    return () => unsubscribe();
  }, [firebaseUser]);

  // --- FUNÇÕES DE EXPORTAÇÃO (ADMIN) ---
  const handleExportCSV = async () => {
    if (!firebaseUser) return;
    setExporting(true);
    try {
      const q = query(
        collection(db, 'artifacts', appId, 'public', 'data', 'opportunities'),
        orderBy('createdAt', 'desc')
      );
      const querySnapshot = await getDocs(q);
      
      // Cabeçalho do CSV
      let csvContent = "data:text/csv;charset=utf-8,";
      csvContent += "Data,Vendedor,Cliente,Telefone,Email,Cidade,UF,Tipo,Oficina,Foco Oficina,Veiculo,Peca,Origem,Vendeu?,Valor,Motivo Perda,Peca Faltante,Obs\n";

      querySnapshot.forEach((doc) => {
        const d = doc.data();
        const date = d.createdAt ? new Date(d.createdAt.seconds * 1000).toLocaleDateString('pt-BR') : '';
        
        // Tratamento de campos para evitar quebra do CSV (remover vírgulas internas)
        const clean = (txt) => txt ? `"${txt.toString().replace(/"/g, '""')}"` : "";

        const row = [
          date,
          clean(d.vendedorNome), // Nome do vendedor logado
          clean(d.clienteNome),
          clean(d.clienteTelefone),
          clean(d.clienteEmail),
          clean(d.clienteCidade),
          clean(d.clienteUF),
          clean(d.tipoCliente),
          clean(d.oficinaNome),
          clean(d.oficinaFoco),
          clean(d.veiculoModelo),
          clean(d.pecaProcurada),
          clean(d.origem),
          d.houveVenda ? "SIM" : "NÃO",
          clean(d.valorVenda),
          clean(d.motivoPerda),
          clean(d.pecaFaltante),
          clean(d.observacoes)
        ].join(",");
        csvContent += row + "\n";
      });

      const encodedUri = encodeURI(csvContent);
      const link = document.createElement("a");
      link.setAttribute("href", encodedUri);
      link.setAttribute("download", `kobber_leads_${new Date().toLocaleDateString('pt-BR').replace(/\//g,'-')}.csv`);
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);

    } catch (err) {
      console.error(err);
      alert("Erro ao exportar dados");
    } finally {
      setExporting(false);
    }
  };

  // --- HANDLERS DO FORMULÁRIO ---
  const handleChange = (field, value) => {
    setFormData(prev => ({ ...prev, [field]: value }));
  };

  const handleCurrencyChange = (e) => {
    const rawValue = e.target.value.replace(/\D/g, "");
    const formatted = (parseFloat(rawValue) / 100).toLocaleString("pt-BR", { minimumFractionDigits: 2 });
    handleChange('valorVenda', rawValue ? formatted : '');
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    if (!firebaseUser || !appUser) return;
    setLoading(true);

    try {
      if (!formData.clienteNome || formData.houveVenda === null) {
        alert("Preencha Nome do Cliente e Resultado da Venda.");
        setLoading(false);
        return;
      }

      await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'opportunities'), {
        ...formData,
        vendedorNome: appUser.name, // Salva quem está logado
        vendedorEmail: appUser.email,
        createdAt: serverTimestamp(),
        userId: firebaseUser.uid
      });

      setSuccessMsg('Registro salvo com sucesso!');
      setFormData(initialFormState);
      setTimeout(() => setSuccessMsg(''), 3000);
      window.scrollTo({ top: 0, behavior: 'smooth' });

    } catch (error) {
      console.error("Erro ao salvar:", error);
      alert("Erro ao salvar registro.");
    } finally {
      setLoading(false);
    }
  };

  // --- RENDERIZAÇÃO ---

  if (!firebaseUser) return <div className="flex items-center justify-center h-screen bg-slate-100">Carregando sistema...</div>;

  // Se não estiver "logado" na aplicação (perfil escolhido), mostra Login
  if (!appUser) {
    return <LoginScreen onLogin={setAppUser} />;
  }

  return (
    <div className="min-h-screen bg-slate-100 font-sans text-slate-800 pb-10">
      
      {/* HEADER */}
      <header className="bg-blue-900 text-white shadow-lg sticky top-0 z-50">
        <div className="max-w-4xl mx-auto px-4 py-3 flex justify-between items-center">
          <div className="flex items-center gap-2">
            <div className="bg-white p-1.5 rounded text-blue-900">
              <Car size={24} />
            </div>
            <div>
              <h1 className="text-xl font-bold leading-none tracking-tight">KOBBER</h1>
              <span className="text-xs text-blue-200 font-medium tracking-widest">CRM</span>
            </div>
          </div>
          
          <div className="flex items-center gap-4">
            <div className="text-right hidden sm:block">
              <p className="text-xs text-blue-300">Logado como</p>
              <div className="flex items-center gap-1">
                {appUser.role === 'admin' && <Shield size={12} className="text-yellow-400"/>}
                <p className="text-sm font-semibold">{appUser.name}</p>
              </div>
            </div>
            <button 
              onClick={() => setAppUser(null)}
              className="p-2 bg-blue-800 hover:bg-blue-700 rounded-lg transition-colors"
              title="Sair"
            >
              <LogOut size={18} />
            </button>
          </div>
        </div>
      </header>

      {/* PAINEL ADMINISTRATIVO (Só aparece para Admin) */}
      {appUser.role === 'admin' && (
        <div className="bg-slate-800 text-white py-4 shadow-inner">
          <div className="max-w-4xl mx-auto px-4 flex flex-col sm:flex-row justify-between items-center gap-4">
            <div className="flex items-center gap-2">
              <Shield className="text-yellow-400" />
              <div>
                <h2 className="font-bold">Painel Administrativo</h2>
                <p className="text-xs text-slate-400">Controle de dados e exportação</p>
              </div>
            </div>
            <button
              onClick={handleExportCSV}
              disabled={exporting}
              className="flex items-center gap-2 bg-green-600 hover:bg-green-700 px-4 py-2 rounded-lg font-bold text-sm transition-all shadow-lg active:scale-95 disabled:opacity-50"
            >
              {exporting ? 'Gerando...' : (
                <>
                  <Download size={18} />
                  BAIXAR LISTA DE LEADS (.CSV)
                </>
              )}
            </button>
          </div>
        </div>
      )}

      <main className="max-w-3xl mx-auto px-4 mt-6">
        
        {successMsg && (
          <div className="mb-6 p-4 bg-green-100 border border-green-400 text-green-800 rounded-lg flex items-center gap-3 shadow-sm animate-pulse">
            <CheckCircle className="w-6 h-6" />
            <span className="font-bold">{successMsg}</span>
          </div>
        )}

        <form onSubmit={handleSubmit} className="bg-white rounded-xl shadow-md p-6 border-t-4 border-blue-600">
          
          <div className="flex flex-col sm:flex-row justify-between items-start sm:items-center mb-4">
            <h2 className="text-2xl font-bold text-slate-800 mb-2 sm:mb-0">Novo Registro</h2>
            <div className="text-sm text-slate-500 bg-slate-100 px-3 py-1 rounded-full">
               Vendedor: <strong>{appUser.name}</strong>
            </div>
          </div>

          {/* 1. CLIENTE */}
          <SectionTitle icon={Briefcase} title="1. Dados do Cliente" />
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            <div className="md:col-span-2">
              <label className="block text-sm font-medium text-slate-700 mb-1">Nome do Cliente *</label>
              <input 
                required
                type="text" 
                className="w-full p-3 border border-slate-300 rounded-lg focus:ring-2 focus:ring-blue-500 outline-none"
                placeholder="Ex: João da Silva"
                value={formData.clienteNome}
                onChange={(e) => handleChange('clienteNome', e.target.value)}
              />
            </div>
            
            <div>
              <label className="block text-sm font-medium text-slate-700 mb-1 flex items-center gap-1">
                <Phone size={14}/> WhatsApp / Telefone
              </label>
              <input 
                type="tel" 
                className="w-full p-3 border border-slate-300 rounded-lg focus:ring-2 focus:ring-blue-500 outline-none"
                placeholder="(00) 00000-0000"
                value={formData.clienteTelefone}
                onChange={(e) => handleChange('clienteTelefone', e.target.value)}
              />
            </div>

            <div>
              <label className="block text-sm font-medium text-slate-700 mb-1">Email</label>
              <input 
                type="email" 
                className="w-full p-3 border border-slate-300 rounded-lg focus:ring-2 focus:ring-blue-500 outline-none"
                placeholder="cliente@email.com"
                value={formData.clienteEmail}
                onChange={(e) => handleChange('clienteEmail', e.target.value)}
              />
            </div>

            <div className="md:col-span-2 flex gap-3">
              <div className="flex-grow">
                <label className="block text-sm font-medium text-slate-700 mb-1">Cidade</label>
                <input 
                  type="text" 
                  className="w-full p-3 border border-slate-300 rounded-lg focus:ring-2 focus:ring-blue-500 outline-none"
                  placeholder="Ex: Santa Maria"
                  value={formData.clienteCidade}
                  onChange={(e) => handleChange('clienteCidade', e.target.value)}
                />
              </div>
              <div className="w-24">
                <label className="block text-sm font-medium text-slate-700 mb-1">UF *</label>
                <select 
                  className="w-full p-3 border border-slate-300 rounded-lg bg-white focus:ring-2 focus:ring-blue-500 outline-none"
                  value={formData.clienteUF}
                  onChange={(e) => handleChange('clienteUF', e.target.value)}
                >
                  {ESTADOS_FOCO.map(uf => <option key={uf} value={uf}>{uf}</option>)}
                </select>
              </div>
            </div>
          </div>

          {/* 2. QUALIFICAÇÃO (OFICINA VS CONSUMIDOR) */}
          <SectionTitle icon={Wrench} title="2. Tipo de Cliente" />
          <div className="mb-4">
            <div className="flex gap-4 mb-4">
              <label className={`flex-1 cursor-pointer border rounded-lg p-4 flex items-center justify-center gap-2 transition-all ${formData.tipoCliente === 'Consumidor Final' ? 'bg-blue-50 border-blue-500 text-blue-700 ring-2 ring-blue-500' : 'hover:bg-slate-50'}`}>
                <input 
                  type="radio" 
                  name="tipoCliente" 
                  className="hidden" 
                  checked={formData.tipoCliente === 'Consumidor Final'} 
                  onChange={() => handleChange('tipoCliente', 'Consumidor Final')}
                />
                <User size={20} />
                <span className="font-semibold text-sm sm:text-base">Consumidor Final</span>
              </label>

              <label className={`flex-1 cursor-pointer border rounded-lg p-4 flex items-center justify-center gap-2 transition-all ${formData.tipoCliente === 'Oficina' ? 'bg-orange-50 border-orange-500 text-orange-700 ring-2 ring-orange-500' : 'hover:bg-slate-50'}`}>
                <input 
                  type="radio" 
                  name="tipoCliente" 
                  className="hidden" 
                  checked={formData.tipoCliente === 'Oficina'} 
                  onChange={() => handleChange('tipoCliente', 'Oficina')}
                />
                <Wrench size={20} />
                <span className="font-semibold text-sm sm:text-base">Oficina / Mecânico</span>
              </label>
            </div>

            {formData.tipoCliente === 'Oficina' && (
              <div className="bg-orange-50 p-4 rounded-lg border border-orange-200 animate-fadeIn">
                <div className="mb-3">
                  <label className="block text-sm font-medium text-orange-800 mb-1">Nome da Oficina</label>
                  <input 
                    type="text" 
                    className="w-full p-2 border border-orange-300 rounded bg-white focus:ring-2 focus:ring-orange-500 outline-none"
                    value={formData.oficinaNome}
                    onChange={(e) => handleChange('oficinaNome', e.target.value)}
                  />
                </div>
                <div>
                  <label className="block text-sm font-medium text-orange-800 mb-2">Foco Principal / Serviços (Descreva)</label>
                  <input 
                    type="text"
                    className="w-full p-2 border border-orange-300 rounded bg-white focus:ring-2 focus:ring-orange-500 outline-none"
                    placeholder="Ex: Suspensão, Câmbio automático, Retífica..."
                    value={formData.oficinaFoco}
                    onChange={(e) => handleChange('oficinaFoco', e.target.value)} 
                  />
                </div>
              </div>
            )}
          </div>

          {/* 3. OPORTUNIDADE */}
          <SectionTitle icon={Search} title="3. Oportunidade & Origem" />
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4 mb-4">
            <div>
              <label className="block text-sm font-medium text-slate-700 mb-1">Peça Procurada *</label>
              <input 
                type="text" 
                className="w-full p-3 border border-slate-300 rounded-lg focus:ring-2 focus:ring-blue-500 outline-none"
                placeholder="Ex: Motor Gol G5"
                value={formData.pecaProcurada}
                onChange={(e) => handleChange('pecaProcurada', e.target.value)}
              />
            </div>
            <div>
              <label className="block text-sm font-medium text-slate-700 mb-1">Veículo (Modelo/Ano)</label>
              <input 
                type="text" 
                className="w-full p-3 border border-slate-300 rounded-lg focus:ring-2 focus:ring-blue-500 outline-none"
                placeholder="Ex: Gol 2010"
                value={formData.veiculoModelo}
                onChange={(e) => handleChange('veiculoModelo', e.target.value)}
              />
            </div>
          </div>
          
          <div className="mb-4">
             <label className="block text-sm font-medium text-slate-700 mb-2">Como conheceu a Kobber?</label>
             <SelectButton 
                options={ORIGENS_CLIENTE} 
                selected={formData.origem} 
                onSelect={(val) => handleChange('origem', val)} 
              />
          </div>

          {/* 4. RESULTADO (CRÍTICO) */}
          <SectionTitle icon={DollarSign} title="4. Resultado do Atendimento" />
          
          <div className="flex gap-4 mb-6">
            <button
              type="button"
              onClick={() => handleChange('houveVenda', true)}
              className={`flex-1 p-5 rounded-xl border-2 flex flex-col items-center gap-2 transition-all ${
                formData.houveVenda === true 
                ? 'bg-green-600 border-green-600 text-white shadow-lg scale-105' 
                : 'border-slate-200 text-slate-500 hover:border-green-400 hover:text-green-600'
              }`}
            >
              <CheckCircle size={32} />
              <span className="font-bold text-lg">VENDA FEITA</span>
            </button>

            <button
              type="button"
              onClick={() => handleChange('houveVenda', false)}
              className={`flex-1 p-5 rounded-xl border-2 flex flex-col items-center gap-2 transition-all ${
                formData.houveVenda === false 
                ? 'bg-red-600 border-red-600 text-white shadow-lg scale-105' 
                : 'border-slate-200 text-slate-500 hover:border-red-400 hover:text-red-600'
              }`}
            >
              <XCircle size={32} />
              <span className="font-bold text-lg">NÃO VENDEU</span>
            </button>
          </div>

          {/* CONDICIONAL: SE VENDEU */}
          {formData.houveVenda === true && (
            <div className="bg-green-50 p-6 rounded-xl border border-green-200 mb-6 animate-slideIn">
              <label className="block text-green-800 font-bold mb-2">Valor da Venda (R$)</label>
              <div className="relative">
                <span className="absolute left-4 top-3 text-green-700 font-bold">R$</span>
                <input 
                  type="text" 
                  className="w-full pl-12 pr-4 py-3 text-xl font-bold text-green-800 border-2 border-green-300 rounded-lg focus:outline-none focus:border-green-600 bg-white"
                  placeholder="0,00"
                  value={formData.valorVenda}
                  onChange={handleCurrencyChange}
                />
              </div>
            </div>
          )}

          {/* CONDICIONAL: SE NÃO VENDEU */}
          {formData.houveVenda === false && (
            <div className="bg-red-50 p-6 rounded-xl border border-red-200 mb-6 animate-slideIn">
              <label className="block text-red-800 font-bold mb-3">Motivo Principal da Perda</label>
              <SelectButton 
                options={MOTIVOS_PERDA} 
                selected={formData.motivoPerda} 
                onSelect={(val) => handleChange('motivoPerda', val)} 
              />
              
              {formData.motivoPerda === 'Falta de Estoque' && (
                 <div className="mt-4">
                   <label className="block text-sm font-bold text-red-700 mb-1">Qual peça faltou? (Lista de Espera)</label>
                   <input 
                    type="text" 
                    className="w-full p-2 border border-red-300 rounded bg-white"
                    placeholder="Especifique a peça para compras"
                    value={formData.pecaFaltante}
                    onChange={(e) => handleChange('pecaFaltante', e.target.value)}
                  />
                 </div>
              )}

               {formData.motivoPerda === 'Outro' && (
                 <div className="mt-4">
                   <label className="block text-sm font-bold text-red-700 mb-1">Especifique o motivo</label>
                   <input 
                    type="text" 
                    className="w-full p-2 border border-red-300 rounded bg-white"
                    value={formData.observacoes}
                    onChange={(e) => handleChange('observacoes', e.target.value)}
                  />
                 </div>
              )}
            </div>
          )}

          <SectionTitle icon={ClipboardList} title="Observações / Próximo Passo" />
          <textarea 
            className="w-full p-3 border border-slate-300 rounded-lg focus:ring-2 focus:ring-blue-500 outline-none mb-6 h-24 resize-none"
            placeholder="Ex: Cliente ficou de retornar dia 10..."
            value={formData.observacoes}
            onChange={(e) => handleChange('observacoes', e.target.value)}
          />

          <button 
            type="submit" 
            disabled={loading}
            className="w-full bg-blue-700 hover:bg-blue-800 text-white font-bold py-4 rounded-xl shadow-lg transition-colors flex items-center justify-center gap-2 text-lg disabled:opacity-50 disabled:cursor-not-allowed"
          >
            {loading ? 'Salvando...' : (
              <>
                <Save className="w-6 h-6" />
                SALVAR OPORTUNIDADE
              </>
            )}
          </button>

        </form>

        {/* LISTAGEM RECENTE */}
        <div className="mt-10 mb-20">
          <h3 className="text-lg font-bold text-slate-600 mb-4 px-2">Últimos Registros (Conferência)</h3>
          <div className="space-y-3">
            {recentOpps.length === 0 ? (
              <p className="text-slate-400 text-center italic">Nenhum registro hoje ainda.</p>
            ) : (
              recentOpps.map((opp) => (
                <div key={opp.id} className="bg-white p-4 rounded-lg shadow-sm border-l-4 border-slate-300 flex justify-between items-center">
                   <div>
                      <p className="font-bold text-slate-800">{opp.clienteNome}</p>
                      <p className="text-xs text-slate-500">{opp.veiculoModelo} - {opp.pecaProcurada}</p>
                      <p className="text-xs text-slate-400 mt-1">Vendedor: {opp.vendedorNome}</p>
                   </div>
                   <div className="text-right">
                      {opp.houveVenda ? (
                        <span className="bg-green-100 text-green-700 px-2 py-1 rounded text-xs font-bold block mb-1">VENDA</span>
                      ) : (
                        <span className="bg-red-100 text-red-700 px-2 py-1 rounded text-xs font-bold block mb-1">PERDA</span>
                      )}
                      <span className="text-xs text-slate-400">
                        {opp.createdAt ? new Date(opp.createdAt.seconds * 1000).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}) : '...'}
                      </span>
                   </div>
                </div>
              ))
            )}
          </div>
        </div>

      </main>
    </div>
  );
}
